
class ModelInfo(models.Model):
    created_at
    updated_at
    created_by

    class Meta:
        abstract = True


class ViewInfo(models.Model):
    view_couter

    class Meta:
        abstract = True


class Page(ModelInfo):
    uuid
    slug


class ContentTypeVideo(ModelInfo, ViewInfo):
    file_link
    subtitles_file_link


class ContentTypeAudio(ModelInfo, ViewInfo):
    file_link
    bitrate

class ContentTypeText(ModelInfo, ViewInfo):
    text
